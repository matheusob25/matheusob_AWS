with open('actors.csv') as file:
      linhas = file.readlines() 

maior_valor = 0
for linha in linhas[1:]:
    valores = []
    valores_recorrentes = ""
    aspas = False
    
    for char in linha:
        if char == ',' and not aspas:
            valores.append(valores_recorrentes.strip('"'))
            valores_recorrentes = ""
        else:
            valores_recorrentes += char
        
        if char == '"':
            aspas = not aspas
    
    valores.append(valores_recorrentes.strip('"'))
    numero = int(valores[2])
    if numero > maior_valor:
        maior_valor = numero
        nome_maior_valor = valores[0] 
    
print(f'Ator com maior n√∫mero de filmes: {nome_maior_valor} com {maior_valor} filmes')
